create table Musuario(
id_usu int(3) not null auto_increment,
nom_usu varchar(20) not null,
appat_usu varchar(20) not null,
apmat_usu varchar(20) not null,
password_usu varchar(20) not null,
PRIMARY KEY(id_usu),
KEY(nom_usu));

create table Cpuntajes(
id_puntaje int(3) not null auto_increment,
nom_usu varchar(20) not null,
lecciones_completadas int(20) not null,
puntaje int(10) not null,
PRIMARY KEY(id_puntaje),
KEY(nom_usu));

create table Cimc(
id_imc int(3) not null auto_increment,
nom_usu varchar(20),
altura varchar(20) not null,
peso varchar(20) not null,
imc varchar(20) not null,
PRIMARY KEY(id_imc),
KEY(altura));
